(defn create-temp-csv-path
  ([]
   (let [file (doto (File/createTempFile "test_" ".csv")
                (.deleteOnExit))]
     (.getPath file)))
  ([content]
   (let [path (create-temp-csv-path)]
     (spit path content)
     path)))
#+END_SRC
